    <script>
            //janeiro
            var total_jan = document.getElementById("id_total_mes");
            var gerente = document.getElementById("id_getente");
            var val_gerente = document.getElementById("id_gerente_valor_fixado");
            function efetuaCalculo(){

                total_jan.innerHTML =parseInt(gerente.value)*parseInt(val_gerente.value);
            }
        gerente.addEventListener("keyup",efetuaCalculo);
        val_gerente.addEventListener("keyup",efetuaCalculo);
    </script>
